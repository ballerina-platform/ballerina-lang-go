module $anon.. v 0.0.0;
import ballerina.io v 0.0.0;
counter  int;
inc() -> nil{
  bb0 {
    lock-start "$anon/.:counter" GOTO bb1;
  }
  bb1 {
    %2 = ConstantLoad 1
    %3 = %2;
    %1 = + counter %3;
    counter = %1;
    lock-end "$anon/.:counter" GOTO bb2;
  }
  bb2 {
    return;
  }
}
main() -> nil{
  bb0 {
    %1 = inc() -> bb1;
  }
  bb1 {
    %2 = inc() -> bb2;
  }
  bb2 {
    %3 = inc() -> bb3;
  }
  bb3 {
    %4 = println(counter) -> bb4;
  }
  bb4 {
    return;
  }
}
