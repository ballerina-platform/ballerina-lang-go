module $anon.. v 0.0.0;
counter  int;
main() -> nil{
  bb0 {
    lock-start "$anon/.:counter" GOTO bb1;
  }
  bb1 {
    %2 = ConstantLoad 1
    %3 = %2;
    %1 = + counter %3;
    counter = %1;
    %4 = ConstantLoad boom
    %5 = newError error(%4)
    lock-end "$anon/.:counter" GOTO bb2;
  }
  bb2 {
    panic %5;
  }
  bb3 {
    return;
  }
}
