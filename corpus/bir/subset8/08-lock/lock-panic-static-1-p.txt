module $anon.. v 0.0.0;
counter  int;
main() -> nil{
  bb0 {
    lock-start "$anon/.:counter" GOTO bb1;
  }
  bb1 {
    PushScopeFrame 5
    %1 = ConstantLoad 1
    %2 = %1;
    %0 = + counter %2;
    counter = %0;
    %3 = ConstantLoad boom
    %4 = newError error(%3)
    (1, %1) = %4;
    PopScopeFrame
    lock-end "$anon/.:counter" GOTO bb2;
  }
  bb2 {
    panic %1;
  }
  bb3 {
    return;
  }
}
